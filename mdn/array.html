var args = [].slice.call(arguments)
var toArray = function() {
    try {
        return Array.prototype.slice.call(arguments);
    } catch(e) {
        var arr = [];
        for (var i = 0,
        len = arguments.length; i < len; i++) {
            //arr.push(s[i]);
            arr[i] = arguments[i]; //据说这样比push快
        }
        return arr;
    }
}
